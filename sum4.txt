class Shape{
public void draw(){
System.out.println("Drawing Shape");
}
}
class Square extends Shape{
public void draw(){
System.out.println("Drawing square");
}
}
class Circle extends Shape{
public void draw(){
System.out.println("Drawing circle");
}
}
public class Sum4
{
public static void main(String[] args) {
Shape a = new Square();
Shape b = new Circle();
a.draw();
b.draw();
}
}
